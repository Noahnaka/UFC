Banco de dados:
- usuarios (formato base) (1)

Table 1("usuarios"):
- idUsuario
- Nome
- Email
- Senha
- Telefone
- Nascimento

Table 2("eventos"):
- idEventos
- nomeEvento
- codigoEvento
- times
- idUsuario(foreign para quem criou)

Table 3("eventosentrado")
-ideventoEntrado
-idUsuario(foreign)
-idEvento(foreign)

Table4("apostaEvento"):
-idApostaEvento
-idEvento(foreign)
-idUsuario(foreign)
-resultado

Table 5("eventosFutebol"):
-idFutebol(int)
-time1(string)
-time2(string)
-resultado(string) (null no começo)

Table 6("apostasFutebol"):
-idAposta
-idFutebol(foreign)
-idUsuario
-resultado

Table 7("eventosUFC"):
-idUFC
-lutador1
-lutador2
-vencedor
-metodo
-rodada

Table 8("apostasUFC"):
-idAposta
-idUsuario
-idUFC(foreign)
-vencedor
-metodo
-rodada

Table 9("eventosBasquete"):
-idBasquete
-time1(string)

-resultado(string) (null no começo)

Table 10 ("apostasBasquete"):
-idAposta
-idBasquete(foreign)
-idUsuario
-resultado
